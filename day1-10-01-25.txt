public class Area

		int length = 5 ;
		int breadth = 10;
		
		int area_rectangle = (length * breadth);
		System.out.println("Area of Ractangle:"+area_rectangle);
		
		//area of triangle
		
		int base = 4;
		int	height = 12;  
		int area_triangle = (base*height)/2;
		System.out.println("Area of Triangle is:"+area_triangle);
		
		//area of square and circle
		
		int side= 4;
		int area_square = side*side;
	    System.out.println("Area of the square="+area_square);  

		//area of circle
	    
	    double pi = 3.14;
	    double radius = 5.0;
	    double area_circle = (pi*radius*radius);
		System.out.println("Area of circle :"+area_circle);
	    
		
	}
}
