package com.demo;

public class ConditionalDemo {
	public static void main(String[]args) {
		int demo=21;
		int result;
		result = ((demo%2) == 0) ? 0:1;    // conditional operator  
		System.out.println("Result : "+ result);
	}

}

package com.demo;

{

		int first = 22;
		int second = 12;
		
		boolean result;
		
					//true		  &&   false--> false

		//because there are two exression
		
		System.out.println("Result of AND operator : "+ result);
		
		result = ((first%2)==1)  || ((second % 2)==1);  //OR operator
		//if any one is true i argee
		System.out.println("Result of OR operator:"+result);
		
		boolean result1 = !result;        //NOt Operator
		System.out.println("Result of NOT operator : "+result);
	}
}

OUTPUT :
	Result of AND operator : true
	Result of OR operator:false

3) Bitwise Operator :

package com.demo;

public class BitwiseDemo {

	public static void main(String[] args) {
		
		//Bitwise AND
		
		int bitone = 101;
		int bitwo = 40;
		
		int result = (bitone & bitwo);
		
		System.out.println("Bitwise And : "+ result);  //Bitwise AND
		
		//Bitwise OR
		
		int result1 = (bitone | bitwo);
		System.out.println("Bitwise OR : " + result1);   //Bitwise OR
	}

}

OUTPUT :
	Bitwise And : 32
	Bitwise OR : 109


4)Shift Operator :

package com.demo;

public class ShiftDemo 
{
	public static void main(String [] args) {
		int shiftone = 100;
		int shiftwo = 200;
		
		int result = (shiftone >> 2);	//right shift
		System.out.println("result of right shift : " +result);
		
		
		int result1 = (shiftwo << 1);	//left shift	
		System.out.println("result of left shift : " +result1);

	}
}


OUTPUT :
	result of right shift : 25
	result of left shift : 400


5) Increment & Decrement :

package com.demo;

public class IncrDecrDemo {

	public static void main(String[] args) {

		int  demo =100;
		int demo1;
		
		demo1= ++demo;
		System.out.println("Pre-Increment : "+demo1);
		
		demo1= --demo;
		System.out.println("Pre-Decrement : "+demo1);
		
		demo1=demo++;
		System.out.println("Post-Increment : "+demo1);
		
		demo1=demo--;
		System.out.println("Post-Decrement : "+demo1);
		
		
		//Hidden Operator
		
		int demo2 = demo+2;
		System.out.println("Final demo2 : "+demo2);
		
		demo2 =  demo2+2;
		System.out.println("Final demo2 : "+demo2);
		
		demo2+= 2;
		System.out.println("Final demo2 : "+demo2);
		
		demo2+=3;
		System.out.println("Final demo2 : "+demo2);
		
	}
}

OUTPUT :
	Pre-Decrement : 100
	Post-Increment : 100
	Post-Decrement : 101
	Final demo2 : 102
	Final demo2 : 104
	Final demo2 : 106
	Final demo2 : 109
